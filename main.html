Data representation of current state of the board:

[ [bR, bKn, bB, bQ, bK, bB, bKn, bR],
[bP, bP, bP, bP, bP, bP, bP, bP],
[ , , , , , , , ],
[ , , , , , , , ],
[ , , , , , , , ],
[ , , , , , , , ],
[wP, wP, wP, wP, wP, wP, wP, wP],
[wR, wKn, wB, wQ, wK, wB, wKn, wR] ]

b & w signify black and white
R = Rook
Kn = Knight
B = Bishop
Q = Queen
P = Pawn
